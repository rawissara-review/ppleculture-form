// Returns real-time seat availability — called by the web form on load and every 30s
function doGet() {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    const totalSeats = 30;
    const registeredCount = Math.max(0, sheet.getLastRow() - 1); // subtract header row
    const remainingSeats = Math.max(0, totalSeats - registeredCount);

    return ContentService.createTextOutput(
      JSON.stringify({
        status: 'ok',
        remainingSeats: remainingSeats,
        registeredCount: registeredCount,
        totalSeats: totalSeats
      })
    ).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(
      JSON.stringify({ status: 'error', message: String(error) })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

// Main function to handle POST requests from the registration form
// NOTE: The web form sends Content-Type: text/plain (required for no-cors browser requests).
// e.postData.contents still contains the JSON string — JSON.parse works correctly.
function doPost(e) {
  try {
    // Parse the incoming JSON data
    const data = JSON.parse(e.postData.contents);

    // Get the active spreadsheet and sheet
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

    // Translate role value to Thai
    const roleTranslation = {
      'developer': 'ผู้พัฒนาเกม / สตูดิโอทำเกม',
      'publisher': 'ผู้จัดจำหน่าย / นำเข้า / โปรโมตเกม',
      'influencer': 'Influencer / สื่อมวลชน',
      'esports': 'บุคลากรด้านอีสปอร์ต',
      'designer': 'Designer / ครีเอทีฟ',
      'general': 'ผู้สนใจทั่วไป / ผู้เล่นเกม',
      'other': 'อื่นๆ'
    };

    const roleDisplay = data.role === 'other'
      ? `อื่นๆ - ${data.otherRole}`
      : roleTranslation[data.role] || data.role;

    // Append the data to the sheet
    sheet.appendRow([
      new Date(), // Timestamp
      data.fullName,
      data.organization,
      roleDisplay,
      data.email,
      data.secondaryContact || '-'
    ]);

    // Optional: Send confirmation email via Resend
    // Uncomment the line below after setting up sendConfirmationEmail function
    // sendConfirmationEmail(data.email, data.fullName);

    // Calculate remaining seats (optional - for dynamic updates)
    const totalSeats = 30;
    const registeredCount = sheet.getLastRow() - 1; // Subtract header row
    const remainingSeats = Math.max(0, totalSeats - registeredCount);

    // Return success response
    return ContentService.createTextOutput(
      JSON.stringify({
        status: 'success',
        message: 'ลงทะเบียนสำเร็จ',
        remainingSeats: remainingSeats
      })
    ).setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    // Return error response
    Logger.log('Error: ' + error.toString());

    return ContentService.createTextOutput(
      JSON.stringify({
        status: 'error',
        message: error.toString()
      })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}
